/TASK - 1

DECLARE MYSTR1, MYSTR2 : STRING 
DECLARE C, N : INTEGER
DECLARE MYCHAR1 : CHAR
DECLARE HAS_ALL : BOOLEAN 

MYSTR1 <- ""
MYSTR2 <- "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
C <- 0
N <- 0
MYCHAR1 <- ""
HAS_ALL <- FALSE 

INPUT "PLEASE ENTER YOUR STRING", MYSTR1 
MYSTR1 <- UPPER(MYSTR1)

FOR C <- 1 TO LEN(MYSTR2)
     MYCHAR1 <- MID(MYSTR1, C, 1)
     N <- LOCATE(MYSTR1, MYCHAR1)
     IF N = 1 THEN 
         N <- N + 1
     ENDIF
NEXT

IF N = 26 THEN 
   HAS_ALL <- TRUE
ENDIF

IF HAS_ALL = FALSE THEN 
   OUTPUT "THE STRING ENTERED DOES NOT HAVE ALL THE ALPHABETS"
ELSE 
   OUTPUT "THE STRING ENTERED DOES HAVE ALL THE ALPHABETS"
ENDIF
